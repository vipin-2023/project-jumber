NestJs Server + MongoDb

1. Install nestjs cli (if not):

npm install -g @nestjs/cli

2. Create a new NestJS project:

nest new my-project

3. Navigate to the Project Directory:

cd my-project

4. Generate a new module:

nest g module module-name

5. Generate a controller:

nest g co module-name

6. Generate a service:

nest g s module-name

7. Install Mongoose Packages:

npm install @nestjs/mongoose mongoose

8.In root 'app.module.ts' file, import the 'MongooseModule' :


    
@Module({
    imports: [
      MongooseModule.forRoot('mongodb://localhost/todo-list', {
        useNewUrlParser: true,
        useUnifiedTopology: true,
      }),
      TaskModule,
    ],
  })

9. task.model.ts :


    
                import * as mongoose from 'mongoose';

                export const TaskSchema = new mongoose.Schema({
                  title: String,
                  description: String,
                  done: Boolean,
                });
                
                export interface Task extends mongoose.Document {
                  id: string;
                  title: string;
                  description: string;
                  done: boolean;
                }

10.task.module.ts :


    
import { Module } from '@nestjs/common';
import { MongooseModule } from '@nestjs/mongoose';
import { TaskController } from './task.controller';
import { TaskService } from './task.service';
import { TaskSchema } from './task.model';

@Module({
  imports: [MongooseModule.forFeature([{ name: 'Task', schema: TaskSchema }])],
  controllers: [TaskController],
  providers: [TaskService],
})
export class TaskModule {}
